workspace {

    model {

        user = person "Użytkownik" {
            description "Analizuje popularność imion w Polsce, przegląda rankingi, mapy oraz dane demograficzne."
        }

        # Systemy zewnętrzne
        gus = softwareSystem "GUS BDL API" {
            description "Bank Danych Lokalnych GUS udostępniający dane o ludności, urbanizacji oraz strukturze demograficznej."
            tags "External System"
        }

        usc = softwareSystem "Portal Dane.gov.pl / Dane USC" {
            description "Zewnętrzny portal danych, z którego pobierany jest oficjalny wykaz imion pierwszych nadanych dzieciom w Polsce (pliki CSV publikowane przez Ministerstwo Cyfryzacji)."
            tags "External System"
        }

        # Nasz system rozbity na kontenery (Containers)
        system = softwareSystem "Trendy Imion w Polsce" {
            
            frontend = container "Frontend Application" {
                description "Aplikacja webowa serwująca interfejs użytkownika, mapy i wykresy statystyczne."
                technology "HTML, JavaScript, Leaflet, Chart.js"
                tags "Web Browser"
            }

            backend = container "Backend API" {
                description "Udostępnia endpointy REST API, realizuje logikę biznesową oraz uruchamia harmonogram synchronizacji danych."
                technology "Python / FastAPI / APScheduler"
                
                # Komponenty wewnątrz Backend API
                namesRouter = component "Names Router" {
                    description "Obsługuje zapytania dotyczące danych demograficznych."
                    technology "FastAPI Router (routers/names.py)"
                }
                
                gusRouter = component "GUS Router" {
                    description "Obsługuje zapytania dotyczące imion."
                    technology "FastAPI Router (routers/gus.py)"
                }

                schemas = component "Schemas" {
                    description "Waliduje dane wejściowe i wyjściowe API."
                }

                scheduler = component "Sync Scheduler" {
                    description "Uruchamia cykliczną synchronizację danych."
                    technology "APScheduler (main.py)"
                }
                
                dbConfig = component "Database Configuration" {
                    description "Zarządza połączeniem, sesjami bazy danych oraz konfiguracją silnika."
                    technology "SQLAlchemy (database.py)"
                }
                
                models = component "Data Models" {
                    description "Definiuje modele danych aplikacji."
                    technology "SQLAlchemy Models (models.py)"
                }
            }

            database = container "Database" {
                description "Przechowuje ustrukturyzowane dane o imionach, ludności oraz wskaźnikach demograficznych."
                technology "PostgreSQL"
                tags "Database"
            }

            ingestion = container "Data Ingestion Scripts" {
                description "Skrypty uruchamiane do pobierania danych demograficznych oraz parsowania plików CSV."
                technology "Python"
            }
        }

        # Relacje między użytkownikiem a kontenerami
        user -> frontend "Przegląda rankingi, mapy i analizuje wykresy" ""
        
        # Relacje wewnątrz systemu (między kontenerami)
        frontend -> backend "Wykonuje zapytania o dane (API requests)" "JSON / HTTP"
        backend -> database "Odpytuje i pobiera dane statystyczne" "SQL"
        ingestion -> database "Zapisuje sparsowane i przetworzone dane" "SQL"

        # Relacje kontenerów z systemami zewnętrznymi
        ingestion -> gus "Pobiera dane demograficzne" "JSON / HTTPS"
        ingestion -> usc "Importuje dane z pobranych plików źródłowych" "Plik CSV / Manual"
        
        # Powiązania komponentów (Niezbędne do wygenerowania widoku Component)
        frontend -> namesRouter "Wykonuje zapytania o statystyki imion" "JSON / HTTP"
        frontend -> gusRouter "Wykonuje zapytania o mapy i demografię" "JSON / HTTP"
        
        namesRouter -> dbConfig "Pobiera sesję bazy danych" "Wywołanie funkcji"
        gusRouter -> dbConfig "Pobiera sesję bazy danych" "Wywołanie funkcji"
        namesRouter -> models "Korzysta z modeli do mapowania" "Import"
        gusRouter -> models "Korzysta z modeli do mapowania" "Import"
        
        # Relacje nowych komponentów (Schemas i Scheduler)
        namesRouter -> schemas "Używa do strukturyzacji odpowiedzi" "Zależność"
        gusRouter -> schemas "Używa do strukturyzacji odpowiedzi" "Zależność"
        scheduler -> ingestion "Wywołuje zadania aktualizacji danych" "Uruchomienie"
        
        dbConfig -> database "Otwiera połączenia i wykonuje zapytania" "SQL"
    }

    views {

        systemContext system "SystemContext" {
            include *
            autoLayout lr
        }

        container system "ContainerContext" {
            include *
            autoLayout lr
        }

        # Widok komponentów dla Backend API
        component backend "ComponentContext" {
            include *
            autoLayout lr
        }

        theme default

        styles {

            element "External System" {
                background #999999
                color #ffffff
                border Dashed
            }

            element "Web Browser" {
                shape WebBrowser
            }

            element "Database" {
                shape Cylinder
            }
            
            element "Component" {
                background #85bbf0
                color #000000
            }
        }
    }
}
